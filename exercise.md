let xhr = new XMLHttpRequest();
xhr.open("GET", "https://restcountries.com/v3.1/all");

xhr.onload = function() {
  if(xhr.status >= 200 && xhr.status < 300) {
    let data = JSON.parse(this.responseText);
    let asianCountries = data.filter((country) => {
